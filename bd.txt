-- 1. Tipos ENUM (PostgreSQL requiere crear los tipos explícitamente)
DO $$ 
BEGIN
    IF NOT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'unidad_medida_type') THEN
        CREATE TYPE unidad_medida_type AS ENUM ('kg', 'tonelada', 'pieza', 'caja', 'manojo');
    END IF;
    IF NOT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'tipo_empresa_type') THEN
        CREATE TYPE tipo_empresa_type AS ENUM ('restaurante', 'supermercado', 'distribuidor', 'catering', 'central_abasto');
    END IF;
    IF NOT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'estado_pedido_type') THEN
        CREATE TYPE estado_pedido_type AS ENUM ('pendiente', 'confirmado', 'en_preparacion', 'enviado', 'entregado', 'cancelado');
    END IF;
END $$;

-- 2. Función para manejar el "ON UPDATE CURRENT_TIMESTAMP" en PostgreSQL
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = NOW();
    RETURN NEW;
END;
$$ language 'plpgsql';

-- ============================================================
-- TABLAS
-- ============================================================

-- TABLA: categorias
CREATE TABLE IF NOT EXISTS categorias (
  id SERIAL PRIMARY KEY,
  nombre VARCHAR(100) NOT NULL UNIQUE,
  descripcion TEXT,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX IF NOT EXISTS idx_categorias_nombre ON categorias (nombre);

-- TABLA: productos
CREATE TABLE IF NOT EXISTS productos (
  id SERIAL PRIMARY KEY,
  nombre VARCHAR(200) NOT NULL,
  categoria_id INT NOT NULL REFERENCES categorias(id) ON DELETE RESTRICT ON UPDATE CASCADE,
  precio_mayoreo DECIMAL(10,2) NOT NULL CHECK (precio_mayoreo >= 0),
  unidad_medida unidad_medida_type NOT NULL DEFAULT 'kg',
  cantidad_disponible DECIMAL(12,2) NOT NULL DEFAULT 0 CHECK (cantidad_disponible >= 0),
  fecha_cosecha DATE,
  fecha_caducidad DATE,
  descripcion TEXT,
  imagen_url VARCHAR(500) DEFAULT '',
  activo BOOLEAN NOT NULL DEFAULT TRUE,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX IF NOT EXISTS idx_productos_nombre ON productos (nombre);
CREATE INDEX IF NOT EXISTS idx_productos_categoria ON productos (categoria_id);
CREATE INDEX IF NOT EXISTS idx_productos_activo ON productos (activo);

-- Trigger para productos updated_at
DROP TRIGGER IF EXISTS trg_update_productos_timestamp ON productos;
CREATE TRIGGER trg_update_productos_timestamp
BEFORE UPDATE ON productos
FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();

-- TABLA: empresas
CREATE TABLE IF NOT EXISTS empresas (
  id SERIAL PRIMARY KEY,
  nombre VARCHAR(200) NOT NULL,
  tipo tipo_empresa_type NOT NULL,
  rfc VARCHAR(13) NOT NULL UNIQUE,
  email VARCHAR(255) NOT NULL UNIQUE,
  password VARCHAR(255) NOT NULL DEFAULT '$2b$10$EixZAYVK1VzKNzQbBZguHeqpZtHGW.6A23.K4.f0sC6X0bB1Q6t2q', -- Hash para 'agrolink123'
  telefono VARCHAR(20) DEFAULT '',
  direccion TEXT DEFAULT '',
  ciudad VARCHAR(100) DEFAULT '',
  estado VARCHAR(100) DEFAULT '',
  activo BOOLEAN NOT NULL DEFAULT TRUE,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX IF NOT EXISTS idx_empresas_tipo ON empresas (tipo);
CREATE INDEX IF NOT EXISTS idx_empresas_rfc ON empresas (rfc);

-- TABLA: api_keys
CREATE TABLE IF NOT EXISTS api_keys (
  id SERIAL PRIMARY KEY,
  empresa_id INT NOT NULL REFERENCES empresas(id) ON DELETE CASCADE ON UPDATE CASCADE,
  api_key VARCHAR(100) NOT NULL UNIQUE,
  nombre VARCHAR(200) NOT NULL,
  activo BOOLEAN NOT NULL DEFAULT TRUE,
  ultimo_uso TIMESTAMP WITH TIME ZONE DEFAULT NULL,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX IF NOT EXISTS idx_api_keys_key ON api_keys (api_key);

-- TABLA: pedidos
CREATE TABLE IF NOT EXISTS pedidos (
  id SERIAL PRIMARY KEY,
  empresa_id INT NOT NULL REFERENCES empresas(id) ON DELETE RESTRICT ON UPDATE CASCADE,
  estado estado_pedido_type NOT NULL DEFAULT 'pendiente',
  total DECIMAL(12,2) NOT NULL DEFAULT 0,
  notas TEXT,
  fecha_entrega_estimada DATE,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX IF NOT EXISTS idx_pedidos_empresa ON pedidos (empresa_id);
CREATE INDEX IF NOT EXISTS idx_pedidos_estado ON pedidos (estado);

-- Trigger para pedidos updated_at
DROP TRIGGER IF EXISTS trg_update_pedidos_timestamp ON pedidos;
CREATE TRIGGER trg_update_pedidos_timestamp
BEFORE UPDATE ON pedidos
FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();

-- TABLA: detalle_pedidos
CREATE TABLE IF NOT EXISTS detalle_pedidos (
  id SERIAL PRIMARY KEY,
  pedido_id INT NOT NULL REFERENCES pedidos(id) ON DELETE CASCADE ON UPDATE CASCADE,
  producto_id INT NOT NULL REFERENCES productos(id) ON DELETE RESTRICT ON UPDATE CASCADE,
  cantidad DECIMAL(10,2) NOT NULL CHECK (cantidad > 0),
  precio_unitario DECIMAL(10,2) NOT NULL CHECK (precio_unitario >= 0),
  subtotal DECIMAL(12,2) NOT NULL CHECK (subtotal >= 0)
);

CREATE INDEX IF NOT EXISTS idx_detalle_pedido ON detalle_pedidos (pedido_id);

-- VISTAS

CREATE OR REPLACE VIEW v_productos_completos AS
SELECT
  p.id,
  p.nombre,
  c.nombre AS categoria,
  p.precio_mayoreo,
  p.unidad_medida,
  p.cantidad_disponible,
  p.fecha_cosecha,
  p.fecha_caducidad,
  p.descripcion,
  p.activo,
  p.created_at,
  p.updated_at
FROM productos p
JOIN categorias c ON p.categoria_id = c.id;

CREATE OR REPLACE VIEW v_pedidos_completos AS
SELECT
  pe.id,
  e.nombre AS empresa,
  e.tipo AS tipo_empresa,
  pe.estado,
  pe.total,
  pe.notas,
  pe.fecha_entrega_estimada,
  pe.created_at,
  pe.updated_at,
  (SELECT COUNT(*) FROM detalle_pedidos dp WHERE dp.pedido_id = pe.id) AS total_items
FROM pedidos pe
JOIN empresas e ON pe.empresa_id = e.id;

CREATE OR REPLACE VIEW v_ventas_por_empresa AS
SELECT
  e.id AS empresa_id,
  e.nombre AS empresa,
  e.tipo,
  COUNT(p.id) AS total_pedidos,
  SUM(CASE WHEN p.estado = 'entregado' THEN p.total ELSE 0 END) AS total_entregado,
  SUM(CASE WHEN p.estado IN ('pendiente', 'confirmado', 'en_preparacion', 'enviado') THEN p.total ELSE 0 END) AS total_en_proceso
FROM empresas e
LEFT JOIN pedidos p ON e.id = p.empresa_id
GROUP BY e.id, e.nombre, e.tipo;

CREATE OR REPLACE VIEW v_productos_mas_vendidos AS
SELECT
  pr.id AS producto_id,
  pr.nombre AS producto,
  c.nombre AS categoria,
  SUM(dp.cantidad) AS total_vendido,
  pr.unidad_medida,
  SUM(dp.subtotal) AS ingreso_total
FROM detalle_pedidos dp
JOIN productos pr ON dp.producto_id = pr.id
JOIN categorias c ON pr.categoria_id = c.id
JOIN pedidos pe ON dp.pedido_id = pe.id
WHERE pe.estado != 'cancelado'
GROUP BY pr.id, pr.nombre, c.nombre, pr.unidad_medida
ORDER BY total_vendido DESC;

-- Funcion: Crear un nuevo pedido
CREATE OR REPLACE FUNCTION sp_crear_pedido(
  p_empresa_id INT,
  p_notas TEXT,
  p_fecha_entrega DATE
)
RETURNS INT AS $$
DECLARE
  v_pedido_id INT;
BEGIN
  INSERT INTO pedidos (empresa_id, estado, total, notas, fecha_entrega_estimada)
  VALUES (p_empresa_id, 'pendiente', 0, p_notas, p_fecha_entrega)
  RETURNING id INTO v_pedido_id;
  
  RETURN v_pedido_id;
END;
$$ LANGUAGE plpgsql;

-- Funcion: Cambiar estado
CREATE OR REPLACE FUNCTION sp_cambiar_estado_pedido(
  p_pedido_id INT,
  p_nuevo_estado estado_pedido_type
)
RETURNS VOID AS $$
BEGIN
  UPDATE pedidos
  SET estado = p_nuevo_estado
  WHERE id = p_pedido_id;
END;
$$ LANGUAGE plpgsql;

-- Trigger: Actualizar stock cuando se entrega un pedido
CREATE OR REPLACE FUNCTION fn_actualizar_stock_entrega()
RETURNS TRIGGER AS $$
BEGIN
  IF NEW.estado = 'entregado' AND OLD.estado != 'entregado' THEN
    UPDATE productos p
    SET cantidad_disponible = GREATEST(0, p.cantidad_disponible - dp.total_cant)
    FROM (
        SELECT producto_id, SUM(cantidad) as total_cant 
        FROM detalle_pedidos 
        WHERE pedido_id = NEW.id 
        GROUP BY producto_id
    ) dp
    WHERE p.id = dp.producto_id;
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trg_actualizar_stock_entrega ON pedidos;
CREATE TRIGGER trg_actualizar_stock_entrega
AFTER UPDATE ON pedidos
FOR EACH ROW EXECUTE FUNCTION fn_actualizar_stock_entrega();
